56      /**
57       * Kutsuu yläluokkansa metodia repaint() ja renderöi ruudun sisällön uudestaan.
58       * @param g 
59       */
60      @Override
61      public void paintComponent(Graphics g) {
62          Graphics2D g2d = (Graphics2D) g;    // Java 2D
63          super.paintComponent(g2d);        
64          setBackground(Color.BLACK);
65          g2d.setRenderingHint(RenderingHints.KEY_ANTIALIASING, RenderingHints.VALUE_ANTIALIAS_ON);
66          g2d.setColor(Color.WHITE);
67          if (actors != null) {
68              for (Actor x : actors) {
69                  g2d.fillRect(x.getX() + 7, x.getY() + 5, x.getWidth() - 12, x.getHeight() - (int) (x.getHeight() * 0.8));
70                  g2d.drawRect(x.getX(), x.getY(), x.getWidth(), x.getHeight());
71              }
72          }
73          if (player != null) {
74              g2d.drawRect(player.getX(), player.getY(), player.getWidth(), player.getHeight());
75              //g2d.fillRect(player.getX() + 7, player.getY() + 7, player.getWidth() - 17, player.getHeight() - 15);
76          }
77          g2d.setFont(myFont);
78          score = "" + (GameMain.score / 10) + "0";
79          textWidth = metric.stringWidth(score);
80          g2d.drawString(score, GameMain.width - textWidth - 10, 50);
81          if (GameMain.highscore == 0) {
82              g2d.drawString("0", 10, 50);
83          } else {
84              g2d.drawString("" + (GameMain.highscore / 10) + "0", 10, 50);
85          }
86          /*
87          for (Integer x : stars) {
88              g2d.fillRect(x / 400, x / 1100, 4, 4);
89          }
90          /*
91          score = "" + GameMain.score;
