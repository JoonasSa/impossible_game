2   package fi.joonassa.impossiblegameproject.listener;
3   
4   import fi.joonassa.impossiblegameproject.actors.Player;
5   import java.awt.event.KeyEvent;
6   import java.awt.event.KeyListener; 
7   /**
8    * Näppäimistönkuuntelija.
9    */
10  public class GameListener implements KeyListener {
11      
12      private boolean jumped;
13      
14      /**
15       * Luo uuden näppäimistönkuuntelija.
16       */
17      public GameListener() {
18          jumped = false;
19      }
20      
21      /**
22       * Palauttaa onko oikeanlainen syöte saatu.
23       * @return totuusarvona true mikäli pelaaja tahtoo hypätä
24       */
25      public boolean getDidPlayerJump() {
26          if (jumped) {
27              jumped = false;
28              return true;
29          }
30          return false;
31      }
32      
33      /**
34       * Kuuntelee näppäimiä ja asettaa player movingUp = true, mikäli painettu näppäin oli 
35       * välilyönti tai ylänuoli.
36       * @param e painettu näppäin.
37       */
38      @Override
39      public void keyPressed(KeyEvent e) {
40          if (Player.canJump && (e.getKeyCode() == KeyEvent.VK_SPACE || e.getKeyCode() == KeyEvent.VK_UP)) {
41              jumped = true;
42              //player.jump();
43          }
44      }
45  
46      /**
47       * Asettaa player movingUp = false, kun näppäimestä päästetään irti.
48       * @param e vapautettu näppäin.
49       */
50      @Override
51      public void keyReleased(KeyEvent e) {
52          jumped = false;
53          //player.drop();
54      }
55          
56      @Override
57      public void keyTyped(KeyEvent e) {
58          
59      }  
60  }
